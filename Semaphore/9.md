semaphore input = 1;//互斥量值为1不用再设置
semaphore a12 = 0, a23 = 0;
semaphore Sb = 0, Sy = 0, Sz = 0;//signal b,y,z，同步b,y,z是否可用

P1(){
    while(1){
        P(input);
        输入数据a;
        V(a12);

        P(Sb);
        x = a + b;
        P(Sy);
        P(Sz);

    }
    
}

    while(1){
        P(a12);

        y = a * b;
        V(Sy);//P3需要y
        V(Sy);//P1需要y，即使先唤醒P1也没事,P1会被之后的P(Sz)阻塞
    }
}
