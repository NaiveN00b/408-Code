```
semaphore empty1 = 10, empty2 = 10;
semaphore full1 = 0, full2 = 0;
semaphore mutex1 = 1, mutex2 = 1;
producerA(){
    while(1){
        生产A;
        P(empty1);
        P(mutex1);
        把A放到货架F1;
        V(mutex1);
        V(full1); 
    }
    
}
producerB(){
    while(1){
        生产B;
        P(empty2);
        P(mutex2);
        把B放到货架F2;
        V(mutex2);
        V(full2);
    }
    
}
consumer(){
    while(1){
        P(full1);
        P(mutex1);
        从货架F1取A;
        V(mutex1);
        V(empty1);

        P(full2);
        P(mutex2);
        从货架F2取B;
        V(mutex2);
        V(empty2);
        将A、B组装成产品;
    }
}
```
